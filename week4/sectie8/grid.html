grid

Container Properties

//Creates a grid

display: grid;

//Direction, establishes main-axis

grid-direction: row || row-reverse || column || columnreverse

//To wrap or not to wrap

grid-wrap: nowrap || wrap || wrapreverse

//Shorthanded (direction wrap)

grid-flow: grid-direction grid-wrap

//Allignement main axis

justify-content: flex-start || flex end || center || space-between || space around || space-evenly

//Allignement cross axis

Allign-items:stretch || grid-start || grid-end || center || baseline

//Allignement lines within the container

align-content: normal || grid-start || grid-end || center || space-between || space-around || space-evenly || stretch

Item properties

//Order of the (grid) html-elements

order: integer (default: 0)

//Ability to grow if necessary

grid-grow: integer (default: 0)

//Ability to shrink if necessary

grid-shrink: positive integer (default: 1)

//Defines default size of the item

grid-basis: lenght || keyword (default auto)

//Shorthanded (grow shrink basis)

grid: grid-grow grid-shrink grid-basis (default 0 1 auto)

//Override default Allignement form allign-items

align-self: auto || grid-start || grid-end || center || baseline || stretch
